                              />
                            </td>

                            {/* Cost Price */}
                            <td className=" px-3 py-2">
                              <input
                                type="number"
                                value={variant.variantCostPrice || ""}
                                onChange={(e) =>
                                  handleVariantChange(
                                    index,
                                    "variantCostPrice",
                                    e.target.value,
                                  )
                                }
                                className="w-24 rounded border px-2 py-1"

                              />

                            <td className=" px-3 py-2">
                              <input
                                type="number"
                                value={variant.variantStock || ""}
                                onChange={(e) =>
                                  handleVariantChange(
                                    index,
                                    "variantStock",

                                  )
                                }
                                className="w-20 rounded border px-2 py-1"

                </div>

          )}

          {step > 1 && (

            >
              Previous
            </button>
          )}

          {step < 4 ? (
            <button
              type="button"
              onClick={() => setStep(step + 1)}
              className="px-6 py-1 bg-[#F6F8F9] border border-[#1C3753] text-[#1C3753] font-medium rounded-md"

              Next
            </button>
          ) : (

            //   Save Product

        </div>
      </form>
    </>
  );
};
